    <script>

        const listPage = document.getElementById('list-page');

        const videoPage = document.getElementById('video-page');

        const ameVideo = document.getElementById('ame-video');


        // 「雨」がタップされたとき

        function playVideo() {

            listPage.style.display = 'none';   // 一覧画面を隠す

            ameVideo.currentTime = 0;          // 動画を最初からにする

        }


        // 「もどる」がタップされたとき

        function backToList() {

            ameVideo.pause();                  // 動画をストップ

            listPage.style.display = 'flex';   // 一覧画面を表示する

            videoPage.style.display = 'none';  // 雨画面を隠す

        }
